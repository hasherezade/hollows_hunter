54bool isAllocationExecutable(std::uint32_t pid, LPVOID baseAddress)
55{
56 bool isExec = false;
57
58 HANDLE hProcess = OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_OPERATION, FALSE, pid);
59 if (hProcess)
60 {
61 bool stop = false;
62 PVOID base = 0;
63 LPVOID addr = baseAddress;
64 MEMORY_BASIC_INFORMATION mbi = { 0 };
65
66 do
67 {
68 if (VirtualQueryEx(hProcess, addr, &mbi, sizeof(MEMORY_BASIC_INFORMATION)) && mbi.AllocationBase)
69 {
70 // Start of the allocation
71 if (!base)
72 {
73 base = mbi.AllocationBase;
74 }
75
76 // Still within?
77 if (base == mbi.AllocationBase)
78 {
79 if (mbi.AllocationProtect & EXECUTABLE_FLAGS || mbi.Protect & EXECUTABLE_FLAGS)
80 {
81 std::cout << "New Executable Section: " << " (" << pid << ") 0x" << std::hex << addr << " Flags=[Alloc: " << mbi.AllocationProtect << " | Now: " << mbi.Protect << "] " << std::dec << std::endl;
82 isExec = true;
83 }
84
85 // Move to next block
86 addr = static_cast<char*>(addr) + mbi.RegionSize;
87 }
88 else
89 stop = true;
90 }
91 else
92 stop = true;
93
94 } while (!stop && !isExec);
95
96 CloseHandle(hProcess);
97 }
98 return isExec;
99}

174bool ETWstart()
175{
176 krabs::kernel_trace trace(L"HollowsHunter");
177
178 krabs::kernel::process_provider processProvider;
179 krabs::kernel::virtual_alloc_provider virtualAllocProvider;
180
181 // Process Start Trigger
182 processProvider.add_on_event_callback([](const EVENT_RECORD& record, const krabs::trace_context& trace_context)
183 {
184 krabs::schema schema(record, trace_context.schema_locator);
185 if (schema.event_opcode() == WINEVENT_OPCODE_START)
186 {
187 krabs::parser parser(schema);
188 std::string filename = parser.parse<std::string>(L"ImageFileName");
189 if (!isWatchedName(filename)) return;
190
191 std::uint32_t pid = parser.parse<std::uint32_t>(L"ProcessId");
192 // New process reset cooldown just in case
193 resetCooldown(pid);
194
195 std::cout << "Scanning New Process: " << filename << " (" << pid << ")" << std::endl;
196 runHHScan(pid);
197 }
198 });
199
200 // Process VirtualAlloc Trigger
201 virtualAllocProvider.add_on_event_callback([](const EVENT_RECORD& record, const krabs::trace_context& trace_context)
202 {
203 krabs::schema schema(record, trace_context.schema_locator);
204 if (schema.event_opcode() == 98) // VirtualAlloc
205 {
206 krabs::parser parser(schema);
207 std::uint32_t targetPid = parser.parse<std::uint32_t>(L"ProcessId");
208
209 if (!isWatchedPid(targetPid)) return;
210
211 if (!isCooldown(targetPid))
212 return;
213
214 bool doScan = false;
215 LPVOID baseAddress = parser.parse<LPVOID>(L"BaseAddress");
216
217 if (targetPid)
218 {
219 doScan = isAllocationExecutable(targetPid, baseAddress);
220 }
221
222 if (doScan)
223 {
224 updateCooldown(targetPid);
225 runHHScan(targetPid);
226 }
227 }
228 });
229
230 bool isOk = true;
231 trace.enable(processProvider);
232 trace.enable(virtualAllocProvider);
233 try {
234 std::cout << "Starting listener..." << std::endl;
235 trace.start();
236 }
237 catch (std::runtime_error& err) {
238 std::cerr << "[ERROR] " << err.what() << std::endl;
239 isOk = false;
240 }
241 return isOk;
242}
