40 bool get_process_info(DWORD processID, bool &isWow64, char* szProcessName, DWORD processNameSize)
41 {
42 HANDLE hProcess = OpenProcess(PROCESS_QUERY_LIMITED_INFORMATION | PROCESS_VM_READ, FALSE, processID);
43 if (!hProcess) {
44 return false;
45 }
46
47 bool is_ok = true;
48 if (szProcessName && processNameSize > 0) {
49 DWORD cbNeeded = 0;
50 HMODULE hMod = nullptr;
51 memset(szProcessName, 0, processNameSize);
52 if (EnumProcessModules(hProcess, &hMod, sizeof(hMod), &cbNeeded)) {
53 GetModuleBaseNameA(hProcess, hMod, szProcessName, processNameSize);
54 } else { is_ok = false; }
55 }
56 isWow64 = is_wow_64(hProcess);
57 CloseHandle(hProcess);

249size_t HHScanner::scanProcesses(HHScanReport &my_report)
250{
251 size_t count = 0;
252 size_t scanned_count = 0;
253 size_t ignored_count = 0;
254
255 HANDLE hProcessSnapShot = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
256 if (hProcessSnapShot == INVALID_HANDLE_VALUE) {
257 const DWORD err = GetLastError();
258 std::cerr << "[-] Could not create modules snapshot. Error: " << std::dec << err << std::endl;
259 return 0;
260 }
261
262 PROCESSENTRY32 pe32 = { 0 };
263 pe32.dwSize = sizeof(PROCESSENTRY32);
264
265 //check all modules in the process, including the main module:
266 if (!Process32First(hProcessSnapShot, &pe32)) {
267 CloseHandle(hProcessSnapShot);
268 std::cerr << "[-] Could not enumerate processes. Error: " << GetLastError() << std::endl;
269 return 0;
270 }
271 do {
272 // scan callback
273 const t_single_scan_status stat = scanNextProcess(pe32.th32ProcessID, pe32.szExeFile, my_report);
274 if (stat == SSCAN_IGNORED) ignored_count++;
275 if (stat == SSCAN_SUCCESS) scanned_count++;
276 count++;
277
278 } while (Process32Next(hProcessSnapShot, &pe32));
279
280 //close the handles
281 CloseHandle(hProcessSnapShot);
282
283 printScanRoundStats(scanned_count, ignored_count);
284 return count;
285}

326t_single_scan_status HHScanner::scanNextProcess(DWORD pid, char* exe_file, HHScanReport &my_report)
327{
328 bool found = false;
329
330 bool is_process_wow64 = false;
331 process_util::get_process_info(pid, is_process_wow64, nullptr, 0);
332
333 const bool check_time = (hh_args.ptimes != TIME_UNDEFINED) ? true : false;
334#ifdef _DEBUG
335 if (check_time) {
336 std::cout << "Init Time: " << std::hex << this->initTime << std::endl;
337 }
338#endif
339 // filter by the time
340 time_t time_diff = 0;
341 if (check_time) { // if the parameter was set
342 const time_t process_time = util::process_start_time(pid);
343 if (process_time == INVALID_TIME) return SSCAN_ERROR0; //skip process if cannot retrieve the time
344
345 // if HH was started after the process
346 if (this->initTime > process_time) {
347 time_diff = this->initTime - process_time;
348 if (time_diff > hh_args.ptimes) return SSCAN_NOT_MATCH; // skip process created before the supplied time
349 }
350 }
351 //filter by the names/PIDs
352 if (hh_args.names_list.size() || hh_args.pids_list.size()) {
353 if (!util::is_searched_name(exe_file, hh_args.names_list) && !util::is_searched_pid(pid, hh_args.pids_list)) {
354 //it is not the searched process, so skip it
355 return SSCAN_NOT_MATCH;
356 }
357 found = true;
358 }
359 if (!found && hh_args.ignored_names_list.size()) {
360 if (util::is_searched_name(exe_file, hh_args.ignored_names_list)) {
361 return SSCAN_IGNORED;
362 }
363 }
364 if (!hh_args.quiet) {
365 std::cout << ">> Scanning PID: " << std::setw(PID_FIELD_SIZE) << std::dec << pid;
366 std::cout << " : " << exe_file;
367
368 if (is_process_wow64) {
369 std::cout << " : 32b";
370 }
371 if (check_time) {
372 std::cout << " : " << time_diff << "s";
373 }
374 std::cout << std::endl;
375 }
376 //perform the scan:
377 pesieve::t_params &pesieve_args = this->hh_args.pesieve_args;
378 pesieve_args.pid = pid;
379
380 pesieve::t_report report = PESieve_scan(pesieve_args);
381 my_report.appendReport(report, exe_file);
382
383 printSingleReport(report);
384 if (report.scanned > 0) {
385 return SSCAN_SUCCESS;
386 }
387 return SSCAN_ERROR1;
388}
