47std::string version_to_str(DWORD version)
48{
49 BYTE *chunks = (BYTE*)&version;
50 std::stringstream stream;
51 stream << std::hex <<
52 (int)chunks[3] << "." <<
53 (int)chunks[2] << "." <<
